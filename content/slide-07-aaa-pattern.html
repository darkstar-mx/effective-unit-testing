AAA Pattern Readability Through Structure

@Test
void testProcess() {
    Service s = new Service();
    s.init();
    assertTrue(s.isValid());

    var result = s.process(new Request(10));
    assertEquals(20, result.val());
}
@Test
void process_ShouldDoubleValue_WhenInputIsPositive() {
    // Arrange
    var service = new Service();
    var request = new Request(10);

    // Act
    var actual = service.process(request);

    // Assert
    assertEquals(new Response(20), actual);
}

Separate Arrange, Act, and Assert explicitly.
