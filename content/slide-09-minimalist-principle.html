void updateEmail_ShouldWork() {
    var user = new User(
        1L, "John", "Doe",
        "123 Main St", "555-1212",
        "1980-01-01"
    );

    service.updateEmail(user, "new@email.com");

    assertEquals("new@email.com", user.getEmail());
}
